export interface AgentHealthSnapshot {

  agent: DiscoveredAgent
  health: CompositeHealth
  lastActivity: SessionActivity | null
  errors: AgentError[]
  channels: ChannelStatus[]
  checkedAt: number
}

export interface DiscoveredAgent {
  id: string; name: string; workspacePath: string; homePath: string
  configPath: string; launchScript: string; model: string
  gatewayPort: number; channels: Record<string, { enabled: boolean }>
  icon?: string
}

export interface AgentError {
  timestamp: string
  type: 'tool_error' | 'rate_limit' | 'channel_error' | 'crash'
  message: string
  source: string
}

export interface ChannelStatus {
  name: string; enabled: boolean; connected: boolean; lastError: string | null
}

export interface CompositeHealth {
  overall: HealthLevel
  dimensions: { process, activity, errors, channels -- each HealthDimension }
}
